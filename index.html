                    int main() {
                        char ch;
                        int c;
                    
                        printf("Enter a key: ");
                        scanf("%c", &ch);
                        c = ch;
                    
                        if (c >= 65 && c <= 90) {
                            printf("Capital letter\n");
                        } 
                        else if (c >= 97 && c <= 122) {
                            printf("Small letter\n");
                        } 
                        else if (c >= 48 && c <= 57) {
                            printf("Digit\n");
                        } 
                        else {
                            printf("Special character\n");
                        }
                    
                        return 0;
                    }
